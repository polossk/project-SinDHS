周期最大为4，证明？归纳法似乎可以说明，嗯。

#include <cstdio>
int ans[10][4] = {
    0, 0, 0, 0,
    1, 1, 1, 1,
    6, 2, 4, 8,
    1, 3, 9, 7,
    6, 4, 6, 4,
    5, 5, 5, 5,
    6, 6, 6, 6,
    1, 7, 9, 3,
    6, 8, 4, 2,
    1, 9, 1, 9};
int t, n;
int main()
{
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d", &n);
        printf("%dn", ans[n % 10][n % 4]);
    }
    return 0;
