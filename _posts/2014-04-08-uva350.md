同样是一个密码生成机，不过这次的密码不再是普通的直接相加，而是使用最基础的线性密码ax+b来生成密码，并且求循环节长度。

所以说暴力什么的最简单粗暴了。

Code Here
{% highlight cpp %}
#include <iostream>
using namespace std;
int table[1000000];
int main()
{
	int z, i, m, tt = 0;
	while (cin >> z >> i >> m >> table[0])
	{
		if (!z && !i && !m && !table[0]) break;
		int t = 0, flag = 0;
		int f = 0;
		while (flag == 0)
		{
			t++;
			table[t] = (table[t-1] * z + i) % m;
			for (int j = 0; j < t; j++)
			{
				if (table[j] == table[t])
				{
					f = j;
					flag = 1;
					break;
				}
			}
		}
		cout << "Case " << ++tt << ": " << t - f << endl;
	}
	return 0;
}{% endhighlight %}
