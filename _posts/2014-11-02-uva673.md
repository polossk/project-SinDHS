char buf[205];
char s[205];

void solve()
{
    int i;
    memset(s, 0, sizeof(s));
    gets(buf);
    int t =0;
    int len = strlen(buf);
    for (i = 0; i < len; i++)
    {
        char ch = buf[i];
        if (ch == '(' || ch == '[') s[t++] = ch;
        else if (ch == ')')
        {
            if (s[t - 1] == '(') s[--t] = '\0';
            else s[t++] = ch;
        }
        else if (ch == ']')
        {
            if (s[t - 1] == '[') s[--t] = '\0';
            else s[t++] = ch;
        }
    }
    if (s[0] == '\0') puts(aye);
    else puts(nay);
}

int main()
{
    int t = nextInt(); scanf(" ");
    while (t--) solve();
    return 0;
}
