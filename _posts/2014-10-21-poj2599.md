const int MaxN = 1024;
bool g[MaxN][MaxN];
int n, k, res;
bool vis[MaxN];

bool dfs(int m)
{
	for (int i = 1; i <= n; i++)
		if (g[m][i] && !vis[i])
	{
		vis[m] = true;
		if (!dfs(i))
		{
			vis[m] = false;
			res = i;
			return true;
		}
		vis[m] = false;
	}
	return false;
}

void solve()
{
	memset(g, 0, sizeof(g));
	for (int i = 1; i < n; i++)
	{
		int x = nextInt();
		int y = nextInt();
		g[x][y] = g[y][x] = 1;
	}
	memset(vis, 0, sizeof(vis));
	bool ans = dfs(k);
	if (ans)
		printf("First player wins flying to airport %d\n", res);
	else puts("First player loses");
}

int main()
{
	while (~scanf("%d%d", &n, &k)) solve();
    return 0;
