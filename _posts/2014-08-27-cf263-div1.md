const int MaxN = 300005;
int n, a[MaxN];

void solve()
{
    for (int i = 0; i < n; i++) a[i] = nextInt();
    sort(a, a + n);
    int64 sum = 0;
    for (int i = 0; i < n - 1; i++)
        sum  = int64(i + 2) * int64(a[i]);
    sum  = int64(n) * int64(a[n - 1]);
    cout << sum << endl;
}

int main()
{
    n = nextInt();
    solve();
    return 0;
