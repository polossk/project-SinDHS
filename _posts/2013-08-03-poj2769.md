#include <iostream>
#include <cstdlib>
using namespace std;
bool p[100001];
int d[1000001];
int main(){
    int n;
    cin>>n;
    while(n--){
        int num,i,j,k;
        cin>>num;
        for(i=1;i<=num;i++)cin>>d[i];
        bool find;
        for(i=1;;i++){
            find=1;
            memset(p,0,sizeof(p));
            for(j=1;j<=num;j++){
                if(p[d[j]%i]){
                    find=0;break;
                }
                p[d[j]%i]=1;
            }
            if(find)break;
        }
        cout<<i<<endl;
    }
    return 0;
}
