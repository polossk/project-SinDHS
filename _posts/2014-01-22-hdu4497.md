#include <cstdio>
using namespace std;

const int maxN = 100005;
int a[maxN], b[maxN];
int G(0), L(0);

void factor(int n, int& tot){
    int temp,i;
    tot =-1;
    for (int i = 2; i * i <= n; ++i){
        if (n%i == 0){
            a[++tot] = i;
            b[tot] = 0;
            while (n % i == 0){
                ++b[tot];
                n /= i;
            }
        }
    }
    if (n != 1){
        a[++tot] = n;
        b[tot] = 1;
    }
}
void solve(){
    scanf("%d%d",&G,&L);
    if (L % G != 0){ printf("0n"); return; }
    L = L / G;
    int tot;
    factor(L, tot);
    long long ans = 1;
    for (int i = 0; i <= tot; ++i) ans *= (6 * b[i]);
    printf("%I64dn",ans);
}
int main(){
    int t; scanf("%d",&t);
    while (t--) solve();
    return 0;
