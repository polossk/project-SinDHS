这道题严格意义上要用高精度，但是其实用double型可以过，应该是数据弱的原因。

Code Here

{% highlight cpp %}
/****
	*@PoloShen
	*Title:UVa 113
	*/
#include <cstdio>
#include <cmath>
using namespace std;
int main(){
    double n, p;
    while (scanf("%lf%lf", &n, &p) != EOF) {
        printf("%.lfn", pow(p, 1/n));
    }
    return 0;
}{% endhighlight %}
