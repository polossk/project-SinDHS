题意是讲把数分成一个个等分的区间，然后输入一系列的数，分别划分到不同的区间上去，随后统计每个区间上数的个数，然后根据区间数最多的那个为基准，进行计算。

/****
	*@PoloShen
	*Title:HDU 4153
	*/
#include  < iostream>
#include  < cstdio>
#include  < cstring>
using namespace std;

int n, w, i;
int v[101], graph[101];
int s, fen;
int max;

void solve(){
    memset(graph, 0, sizeof(graph));
    max = 0;
    for (i = 0; i < n; i++){
        cin >> v[i];
        graph[v[i]/w]++;
        if (max < v[i]) max = v[i];
    }
    fen = max / w;
    max = 0;
    for (i = 0; i <= fen; i++){
        if (graph[i] > max)
            max = graph[i];
    }
    s = max;
    double ans;
    ans = 0.01;
    for (i = 0; i <= fen; i++){
        ans += (i * 1.0 / fen) * (graph[fen-i] * 1.0 / s);
    }
    printf("%.6fn", ans);
}

int main(){
	while(cin >> n >> w, n||w){
		solve();
	}
	return 0;
