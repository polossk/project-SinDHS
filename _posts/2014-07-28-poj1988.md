const int MaxN = 30005;
struct DisjointSet
{
    int fa[MaxN];
    int und[MaxN], tot[MaxN];
    /*
    other data set:
    */
    DisjointSet(int n) { init(n); }
    inline void init(int n)
    {
        fill(und, und + n, 0);
        fill(tot, tot + n, 1);
        for (int i = 0; i < n; i++) fa[i] = i;
    }
    inline int _find(int x)
    {
        if (fa[x] == x) return x;
        int f = _find(fa[x]);
        und[x] += und[fa[x]];
        fa[x] = f;
        return fa[x];
    }
    inline bool Query(int x, int y) { return _find(x) == _find(y); }
    void Merge(int x, int y)
    {
        x = _find(x); y = _find(y);
        if (x == y) return;
        fa[y] = x;
        //other method
        und[y] = tot[x];
        tot[x] += tot[y];
    }
};
DisjointSet data(30005);
int p; char ch;

{
    data.init(30005);

    for (int i = 0; i < p; i++)
    {

}
