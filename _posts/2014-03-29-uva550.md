题目是说，在某些进制下，数字的乘法可能满足轮换性质，简单的说，就比如179487 * 4 = 717948，就好像整个数字进行了右移运算一样。那么如何计算呢？答案是反复的乘法计算。

Code Here
{% highlight cpp %}
//550
#include <cstdio>
int main()
{
    int x, y, z;
    int sum, mod, m, total;
    while (~scanf("%d %d %d", &x, &y, &z))
    {
        if(y != 0)
        {
            total = m = sum = 0;
            mod = y;
            while (sum != y)
            {
                sum = z * mod + m;
                m = sum / x;
                mod = sum % x;
                total++;
            }
            printf("%dn",total);
        }
        else printf("1n");
    }
    return 0;
}{% endhighlight %}
