题意是给一个n x m的矩形，问多少个3 x 3的正方形可以完全覆盖。

答案是(n / 3) * (m / 3)

Code Here
{% highlight cpp %}
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int t, n, m;
    cin >> t;
    while (t--)
    {
        cin >> n >> m;
        cout << (n / 3) * (m / 3) << endl;
    }
    return 0;
}{% endhighlight %}
