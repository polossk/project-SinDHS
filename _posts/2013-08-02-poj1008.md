/****

#include <iostream>
#include <string>
#include <cstring>
#include <cstdio>
using namespace std;
#define DBG 0
#define ShowLine DBG && cout<<__LINE__<<">>| "
#define dout DBG && cout<<__LINE__<<">>| "
#define write(x) #x" = "<<(x)<<", "

string haab_month[19]={"pop","no","zip","zotz","tzec",
                    "xul","yoxkin","mol","chen","yax",
                    "zac","ceh","mac","kankin","muan",
                    "pax","koyab","cumhu","uayet"};
string name_of_day[20]={"imix","ik","akbal","kan","chicchan",
                    "cimi","manik","lamat","muluk","ok",
                    "chuen","eb","ben","ix","mem",
                    "cib","caban","eznab","canac","ahau"};
const int haab=365;
const int tzolkin=260;
int main(){
    int t;
    cin>>t;
    cout<<t<<endl;
    while (t--){
        int date,month_int,year;string month;
        int sday;char ch;
        ShowLine;cin>>date;
        ShowLine;cin>>ch;
        ShowLine;cin>>month;
        for (int i=0;i<19;i++)
            if (haab_month[i]==month){
                month_int=i;
            }
        ShowLine;cin>>year;
        dout<<date<<". "<<month<<" "<<year<<endl;
        int tday=1,tyear,temp_day_last,name=0;
        sday=haab*year+20*month_int+date+1;
        dout<<write(sday)<<endl;
        tyear=sday/tzolkin;temp_day_last=sday%tzolkin;
        if (temp_day_last==0){
            temp_day_last=260;
            tyear--;
        }
        dout<<write(tyear)<<write(temp_day_last)<<endl;
        for (int i=1;i<=temp_day_last;i++){
            if (i==temp_day_last){
                cout<<tday<<" "<<name_of_day[name]<<" "<<tyear<<endl;
            }
            else {
                tday++;if (tday==14)tday=1;
                name++;if (name==20)name=0;
            }
        }
    }
    return 0;
}
