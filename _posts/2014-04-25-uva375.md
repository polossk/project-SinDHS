题目就是这样一直循环下去取内接圆，知道满足r的大小小于0.000001，最后求所有圆的周长。

循环计算下去即可。

Code Here
{% highlight cpp %}
///uva 375
#include <cmath>
#include <cstdio>
using namespace std;

const double pi = 2 * asin(1.0);
double B, H, r, h, R;

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        scanf("%lf%lf", &B, &H);
        R = (H * H + B * B * 0.25);
        R = sqrt(R);
        R = B * H / (2.0 * R + B);
        h = H;
        r = R;
        while (r >= 0.000001)
        {
            h = h - 2.0 * r;
            r = R * h / H;
        }
        printf("%13lfn", (H - h) * pi);
        if (t) printf("n");
    }
}{% endhighlight %}
